import MathBlock from "@/components/MathBlock"
import PlayControl from "@/components/PlayControl"
import { useState } from 'react';

// --- Local state for the PlayControl ---
const DerivativeInteractive = () => {
  const [h, setH] = useState(2.0); // Initial 'h' value
  
  // The derivative formula with the dynamic 'h' value interpolated as text
  const formula = `f'(x) = \\lim_{h \\to 0} \\frac{f(x+h) - f(x)}{h} \\quad \\text{Current h: } ${h.toFixed(2)}`;

  return (
    <>
      <h2>The Limit Definition: Instantaneous Change</h2>
      <p className="text-secondary-text text-xl my-6">
        The derivative measures the slope of a function **at a single point**. We find this by taking the slope of a <span className="text-accent-main font-medium">secant line</span> over a tiny interval $h$, and watching as $h$ approaches zero.
      </p>

      <div className="my-8">
        <MathBlock expression={formula} />
      </div>

      <div className="max-w-xl mx-auto my-12">
          {/* Play Control Component */}
          <PlayControl 
              label="Interval (h)"
              min={0.01} 
              max={5.0} 
              step={0.01} 
              value={h} 
              onChange={setH} 
          />
      </div>

      <p className="text-center text-sm italic text-secondary-text mt-8">
          Note how the fraction becomes more precise as $h$ gets smaller. Eventually, we define this limit to be the slope of the tangent line.
      </p>
    </>
  );
};

<DerivativeInteractive />

# Derivatives: The Foundation of Calculus

The derivative measures **instantaneous change**. It is the most powerful tool in the study of functions.
